# ---------------------------------------------------------------------------
# SIGReg — Sketch Isotropic Gaussian Regularizer (from LeWorldModel)
# Prevents representation collapse by enforcing Gaussian latent distribution.
# Maes, Le Lidec, Scieur, LeCun, Balestriero (2026)
# ---------------------------------------------------------------------------
class SIGReg(nn.Module):
    def __init__(self, knots: int = 17, num_proj: int = 1024):
        super().__init__()
        self.num_proj = num_proj
        t = torch.linspace(0, 3, knots, dtype=torch.float32)
        dt = 3 / (knots - 1)
        weights = torch.full((knots,), 2 * dt, dtype=torch.float32)
        weights[[0, -1]] = dt
        window = torch.exp(-t.square() / 2.0)
        self.register_buffer("t", t)
        self.register_buffer("phi", window)             # Gaussian characteristic function
        self.register_buffer("weights", weights * window)

    def forward(self, embeddings: Tensor) -> Tensor:
        """
        embeddings: (B, T, D) or (N, D)
        Returns: scalar regularization loss
        """
        if embeddings.ndim == 3:
            emb = embeddings.reshape(-1, embeddings.size(-1))
        else:
            emb = embeddings
        N, D = emb.shape

        # Random projections onto unit sphere
        A = torch.randn(D, self.num_proj, device=emb.device, dtype=emb.dtype)
        A = A / A.norm(dim=0, keepdim=True)

        # Empirical characteristic function vs Gaussian target
        proj = emb @ A                                    # (N, num_proj)
        x_t = proj.unsqueeze(-1) * self.t                 # (N, num_proj, knots)
        err = (x_t.cos().mean(0) - self.phi).square() + x_t.sin().mean(0).square()
        return (err @ self.weights).mean() * N

class RMSNorm(nn.Module):
    def __init__(self, eps: float | None = None):
        super().__init__()
        self.eps = eps

    def forward(self, x: Tensor) -> Tensor:
        return F.rms_norm(x, (x.size(-1),), eps=self.eps)


def segsum(x: Tensor) -> Tensor:
    """Stable segment sum for computing cumulative decay in log-space."""
    T = x.size(-1)
    x = x[..., None].repeat(1, 1, 1, T)                     # (..., T, T)
    mask = torch.tril(torch.ones(T, T, device=x.device, dtype=torch.bool), diagonal=-1)
    x = x.masked_fill(~mask, 0)
    x_segsum = torch.cumsum(x, dim=-2)
    mask2 = torch.tril(torch.ones(T, T, device=x.device, dtype=torch.bool), diagonal=0)
    return x_segsum.masked_fill(~mask2, -torch.inf)


class Mamba2SSD(nn.Module):
    """
    Pure-PyTorch Mamba-2 (Structured State Space Duality).
    Based on the minimal SSD implementation from Gu & Dao (2024), Section 6.
    Uses the quadratic (attention-like) form for training — no custom CUDA kernels needed.

    Key difference from Mamba-1: A is scalar-times-identity per head,
    allowing the recurrence to be computed as a structured masked attention.
    """
    def __init__(self, d_model: int, d_state: int = 64, d_conv: int = 4,
                 expand: int = 2, n_heads: int = None):
        super().__init__()
        d_inner = d_model * expand
        # Default: head_dim = d_state, n_heads = d_inner / head_dim
        self.head_dim = d_state
        self.n_heads = n_heads or (d_inner // d_state)
        d_inner = self.n_heads * self.head_dim  # ensure divisible
        self.d_inner = d_inner

        # Input projections: x, z (gate), B, C, dt
        # All packed into one linear for efficiency
        d_proj = d_inner + d_inner + self.n_heads * d_state + self.n_heads * d_state + self.n_heads
        self.in_proj = nn.Linear(d_model, d_proj, bias=False)

        # Short causal conv on the x branch
        self.conv = nn.Conv1d(d_inner, d_inner, d_conv,
                              padding=d_conv - 1, groups=d_inner, bias=True)

        # Learnable log-space decay (one scalar per head)
        self.A_log = nn.Parameter(torch.randn(self.n_heads))

        # Output projection
        self.out_proj = nn.Linear(d_inner, d_model, bias=False)
        nn.init.zeros_(self.out_proj.weight)

        # Normalization before output
        self.norm = RMSNorm()

    def forward(self, x: Tensor) -> Tensor:
        B, L, D = x.shape
        proj = self.in_proj(x)

        # Split projections
        x_raw, z, B_raw, C_raw, dt_raw = proj.split(
            [self.d_inner, self.d_inner, self.n_heads * self.head_dim,
             self.n_heads * self.head_dim, self.n_heads], dim=-1
        )

        # Causal conv on x branch
        x_conv = self.conv(x_raw.transpose(1, 2))[..., :L].transpose(1, 2)
        x_conv = F.silu(x_conv)

        # Reshape to (B, L, n_heads, head_dim) or (B, n_heads, L, ...)
        x_h = x_conv.reshape(B, L, self.n_heads, self.head_dim)
        B_h = B_raw.reshape(B, L, self.n_heads, self.head_dim)
        C_h = C_raw.reshape(B, L, self.n_heads, self.head_dim)

        # Decay: A is negative (stable), dt is softplus-activated
        A = -torch.exp(self.A_log.float())                     # (n_heads,)
        dt = F.softplus(dt_raw.float())                         # (B, L, n_heads)

        # Discretized log-decay per step: log(a) = A * dt
        decay = A[None, None, :] * dt                           # (B, L, n_heads)
        decay = decay.permute(0, 2, 1)                          # (B, n_heads, L)

        # --- Quadratic (attention-like) SSD computation ---
        # This is the "dual form" from the Mamba-2 paper
        # M_{ij} = C_i^T diag(exp(cumsum_decay_{i} - cumsum_decay_{j})) B_j
        decay_cumsum = segsum(decay)                            # (B, n_heads, L, L)

        # QKV-like: Q=C, K=B, V=x, with structured causal mask from decay
        # Compute (C @ B^T) * exp(decay_cumsum), then matmul with (B_diag @ x)
        C_p = C_h.permute(0, 2, 1, 3)                          # (B, n_heads, L, head_dim)
        B_p = B_h.permute(0, 2, 1, 3)                          # (B, n_heads, L, head_dim)
        x_p = x_h.permute(0, 2, 1, 3)                          # (B, n_heads, L, head_dim)

        # Attention-like scores with decay mask
        attn = C_p @ B_p.transpose(-1, -2)                     # (B, n_heads, L, L)
        attn = attn * torch.exp(decay_cumsum)                   # apply causal decay
        y = attn @ x_p                                          # (B, n_heads, L, head_dim)

        # Reshape back and apply gate
        y = y.permute(0, 2, 1, 3).reshape(B, L, self.d_inner)  # (B, L, d_inner)
        y = self.norm(y) * F.silu(z)

        return self.out_proj(y)


class SSMBlock(nn.Module):
    """SSM block with pre-RMSNorm and residual. Uses Mamba-2 CUDA if available, else pure-PyTorch SSD."""
    def __init__(self, d_model: int, d_state: int = 64, d_conv: int = 4, expand: int = 2):
        super().__init__()
        self.norm = RMSNorm()
        if MAMBA_AVAILABLE:
            self.ssm = Mamba2(d_model=d_model, d_state=d_state, d_conv=d_conv, expand=expand)
        else:
            self.ssm = Mamba2SSD(d_model=d_model, d_state=d_state, d_conv=d_conv, expand=expand)

    def forward(self, x: Tensor) -> Tensor:
        return x + self.ssm(self.norm(x))


# ---------------------------------------------------------------------------
# ByteJEPA — the core model
# ---------------------------------------------------------------------------
class ByteJEPA(nn.Module):
    """
    Byte-level JEPA with SSM backbone.

    Data flow:
        bytes → Embedding(256, D) → [SSMBlock × N] → RMSNorm → Predictor MLP
                                                                      ↓
                                                        predicted next-byte embedding
    Training:
        loss = MSE(predicted, byte_embed(next_byte)) + λ · SIGReg(hidden_states)

    Eval (energy-based scoring):
        logits_i = -||predicted - byte_embed(i)||² / τ   for i ∈ {0..255}
        BPB = -mean(log_softmax(logits)[correct_byte]) / ln(2)
    """

    def __init__(self, model_dim: int = 512, num_layers: int = 12,
                 d_state: int = 64, d_conv: int = 4, expand: int = 2,
                 predictor_hidden_mult: int = 2):
        super().__init__()
        self.model_dim = model_dim

        # Shared byte embedding: used for both input and energy-based output scoring
        self.byte_embed = nn.Embedding(256, model_dim)
        nn.init.normal_(self.byte_embed.weight, std=0.02)

        # SSM backbone
        self.blocks = nn.ModuleList([
            SSMBlock(model_dim, d_state, d_conv, expand)
            for _ in range(num_layers)
        ])
        self.norm = RMSNorm()

        # Predictor: maps hidden state h_t → predicted embedding of byte_{t+1}
        pred_hidden = model_dim * predictor_hidden_mult
        self.predictor = nn.Sequential(
            nn.Linear(model_dim, pred_hidden),
            nn.GELU(),
            nn.Linear(pred_hidden, model_dim),
        )
        # Initialize predictor output near zero for stable start
        nn.init.zeros_(self.predictor[-1].weight)
        nn.init.zeros_(self.predictor[-1].bias)

        # Learnable temperature for energy-based scoring
        self.log_temperature = nn.Parameter(torch.zeros(1, dtype=torch.float32))

    def encode(self, byte_ids: Tensor) -> Tensor:
        """byte_ids: (B, T) → hidden states: (B, T, D)"""
        x = self.byte_embed(byte_ids)
        for block in self.blocks:
            x = block(x)
        return self.norm(x)

    def forward(self, input_ids: Tensor, target_ids: Tensor,
                reduction: str = "mean", temperature: float | None = None) -> Tensor:
        """
        Compute energy-based log-likelihood (for eval / BPB calculation).
        input_ids: (B, T), target_ids: (B, T)
        Returns: scalar loss (mean NLL) or per-token losses if reduction="none"
        """
        h = self.encode(input_ids)                    # (B, T, D)
        pred_emb = self.predictor(h)                  # (B, T, D)
        logits = self._energy_logits(pred_emb, temperature)  # (B, T, 256)

        logits_flat = logits.reshape(-1, 256)
        targets_flat = target_ids.reshape(-1)

        if reduction == "none":
            return F.cross_entropy(logits_flat.float(), targets_flat,
                                   reduction="none").reshape(input_ids.shape)
        return F.cross_entropy(logits_flat.float(), targets_flat)

    def compute_jepa_loss(self, input_ids: Tensor, target_ids: Tensor,
                          sigreg: SIGReg, sigreg_lambda: float = 1.0,
                          ce_weight: float = 0.0, jepa_weight: float = 1.0,
                          detach_targets: bool = True) -> tuple[Tensor, dict]:
        """
        Full training loss: JEPA MSE + SIGReg + optional CE.
        input_ids: (B, T), target_ids: (B, T)
        detach_targets: if True, stop-gradient on target embeddings (stable predictor learning).
        """
        h = self.encode(input_ids)                    # (B, T, D)
        pred_emb = self.predictor(h)                  # (B, T, D)

        diagnostics = {}
        loss = torch.zeros((), device=h.device)

        # JEPA prediction loss (MSE in embedding space)
        if jepa_weight > 0:
            target_emb = self.byte_embed(target_ids)  # (B, T, D)
            if detach_targets:
                target_emb = target_emb.detach()
            jepa_loss = (pred_emb - target_emb).pow(2).mean()
            loss = loss + jepa_weight * jepa_loss
            diagnostics["jepa"] = jepa_loss.detach()

        # SIGReg on hidden states (prevents collapse)
        if sigreg_lambda > 0:
            sig_loss = sigreg(h)
            loss = loss + sigreg_lambda * sig_loss
            diagnostics["sigreg"] = sig_loss.detach()

        # CE loss for calibration
        if ce_weight > 0:
            logits = self._energy_logits(pred_emb)
            ce_loss = F.cross_entropy(logits.reshape(-1, 256),
                                      target_ids.reshape(-1))
            loss = loss + ce_weight * ce_loss
            diagnostics["ce"] = ce_loss.detach()

        return loss, diagnostics

    def _energy_logits(self, pred_emb: Tensor,
                       temperature: float | None = None) -> Tensor:
        """
        Energy-based scoring: neg L2² distance from predicted embedding to all 256 byte embeddings.
        pred_emb: (..., D) → logits: (..., 256)
        """
        if temperature is None:
            tau = self.log_temperature.exp().clamp(min=1e-4)
        else:
            tau = temperature
        all_emb = self.byte_embed.weight              # (256, D)
        # Efficient: expand ||a - b||² = ||a||² + ||b||² - 2·a·b
        pred_sq = pred_emb.pow(2).sum(-1, keepdim=True)       # (..., 1)
        emb_sq = all_emb.pow(2).sum(-1, keepdim=True).t()     # (1, 256)
        dot = pred_emb @ all_emb.t()                           # (..., 256)
        dist_sq = pred_sq + emb_sq - 2 * dot                  # (..., 256)
        return -dist_sq / tau


# ---------------------------------------------------------------------------
# Muon optimizer (Newton-Schulz orthogonalized momentum)
# ---------------------------------------------------------------------------
def _ns_orth(G: Tensor, steps: int = 10, eps: float = 1e-7) -> Tensor:
    a, b, c = (3.4445, -4.7750, 2.0315)
    X = G.bfloat16()
    X /= X.norm() + eps
    transposed = G.size(0) > G.size(1)
    if transposed:
        X = X.T
    for _ in range(steps):
        A = X @ X.T
        B = b * A + c * A @ A
        X = a * X + B @ X
    return X.T if transposed else X

ns_orth = _ns_orth   # will be compiled in main()


class Muon(torch.optim.Optimizer):
    def __init__(self, params, lr: float, momentum: float,
                 backend_steps: int, nesterov: bool = True):
        super().__init__(params, dict(lr=lr, momentum=momentum,
                                      backend_steps=backend_steps, nesterov=nesterov))

    @torch.no_grad()
    def step(self, closure=None):
        distributed = dist.is_available() and dist.is_initialized()
        world_size = dist.get_world_size() if distributed else 1
        rank = dist.get_rank() if distributed else 0

        for group in self.param_groups:
            params = group["params"]
            if not params:
                continue
            lr, mom = group["lr"], group["momentum"]
            steps, nesterov = group["backend_steps"], group["nesterov"]

            total = sum(p.numel() for p in params)
            flat = torch.zeros(total, device=params[0].device, dtype=torch.bfloat16)
            cur = 0
            for i, p in enumerate(params):
                if i % world_size == rank and p.grad is not None:
                    g = p.grad
                    state = self.state[p]
                    if "buf" not in state:
                        state["buf"] = torch.zeros_like(g)
                    buf = state["buf"]
                    buf.mul_(mom).add_(g)
                    if nesterov:
                        g = g.add(buf, alpha=mom)
                    g = F.rms_norm(g.float(), (g.size(-1),)).bfloat16()
                    g = ns_orth(g, steps=steps)
                    g *= max(1, g.size(0) / g.size(1)) ** 0.5
                    flat[cur:cur + p.numel()] = g.reshape(-1)
                cur += p.numel()

            if distributed:
                dist.all_reduce(flat, op=dist.ReduceOp.SUM)

            cur = 0
            for p in params:
                g = flat[cur:cur + p.numel()].view_as(p).to(p.dtype)
                p.add_(g, alpha=-lr)
                cur += p.numel()

def load_byte_shard(path: Path) -> Tensor:
    """Load a raw byte shard. Format: just raw uint8 bytes, no header."""
    raw = np.fromfile(path, dtype=np.uint8)
    return torch.from_numpy(raw)


class ByteStream:
    def __init__(self, pattern: str):
        self.files = [Path(p) for p in sorted(glob.glob(pattern))]
        if not self.files:
            raise FileNotFoundError(f"No files for pattern: {pattern}")
        self.file_idx = 0
        self.data = load_byte_shard(self.files[0])
        self.pos = 0

    def _advance(self):
        self.file_idx = (self.file_idx + 1) % len(self.files)
        self.data = load_byte_shard(self.files[self.file_idx])
        self.pos = 0

    def take(self, n: int) -> Tensor:
        chunks = []
        remaining = n
        while remaining > 0:
            avail = self.data.numel() - self.pos
            if avail <= 0:
                self._advance()
                continue
            k = min(remaining, avail)
            chunks.append(self.data[self.pos:self.pos + k])
            self.pos += k
            remaining -= k
        return chunks[0] if len(chunks) == 1 else torch.cat(chunks)


class DistributedByteLoader:
    def __init__(self, pattern: str, rank: int, world_size: int, device: torch.device):
        self.rank, self.world_size, self.device = rank, world_size, device
        self.stream = ByteStream(pattern)

    def next_batch(self, global_tokens: int, seq_len: int,
                   grad_accum_steps: int) -> tuple[Tensor, Tensor]:
        local_tokens = global_tokens // (self.world_size * grad_accum_steps)
        per_rank_span = local_tokens + 1
        chunk = self.stream.take(per_rank_span * self.world_size)
        start = self.rank * per_rank_span
        local = chunk[start:start + per_rank_span]
        local = local.pin_memory().to(self.device, non_blocking=True).to(torch.int64)
        x = local[:-1].reshape(-1, seq_len)
        y = local[1:].reshape(-1, seq_len)
        return x, y

def load_val_bytes(pattern: str, seq_len: int,
                   max_bytes: int = int(os.environ.get("VAL_MAX_TOKENS", 500_000))) -> Tensor:
    files = sorted(glob.glob(pattern))
    assert files, f"No files: {pattern}"
    data = torch.cat([load_byte_shard(Path(p)) for p in files]).contiguous()
    if max_bytes > 0:
        data = data[:max_bytes + 1]
    u = ((data.numel() - 1) // seq_len) * seq_len
    return data[:u + 1]


def eval_val(args, model, rank, world_size, device, grad_accum_steps,
             val_bytes: Tensor, temperature: float = 1.0):
    """Evaluate BPB using energy-based scoring."""
    local_batch_tokens = args.val_batch_size // (world_size * grad_accum_steps)
    local_batch_seqs = max(1, local_batch_tokens // args.train_seq_len)
    total_seqs = (val_bytes.numel() - 1) // args.train_seq_len
    seq_start = (total_seqs * rank) // world_size
    seq_end = (total_seqs * (rank + 1)) // world_size

    loss_sum = torch.zeros((), device=device, dtype=torch.float64)
    byte_count = torch.zeros((), device=device, dtype=torch.float64)

    model.eval()
    with torch.inference_mode():
        for batch_start in range(seq_start, seq_end, local_batch_seqs):
            batch_end = min(batch_start + local_batch_seqs, seq_end)
            raw_start = batch_start * args.train_seq_len
            raw_end = batch_end * args.train_seq_len + 1
            local = val_bytes[raw_start:raw_end].to(device=device, dtype=torch.int64)
            x = local[:-1].reshape(-1, args.train_seq_len)
            y = local[1:].reshape(-1, args.train_seq_len)

            with torch.autocast(device_type="cuda", dtype=torch.bfloat16):
                batch_loss = model(x, y, temperature=temperature).detach()

            n = float(y.numel())
            loss_sum += batch_loss.to(torch.float64) * n
            byte_count += n

    if dist.is_available() and dist.is_initialized():
        for t in (loss_sum, byte_count):
            dist.all_reduce(t, op=dist.ReduceOp.SUM)

    val_loss = loss_sum / byte_count
    # Each prediction is one byte, so BPB = loss_nats / ln(2)
    bpb = val_loss.item() / math.log(2.0)
    model.train()
    return float(val_loss.item()), float(bpb)


def eval_val_sliding(args, model, rank, world_size, device, grad_accum_steps,
                     val_bytes: Tensor, stride: int = 64, temperature: float = 1.0):
    """Sliding window evaluation for better BPB."""
    seq_len = args.train_seq_len
    batch_size = args.sliding_batch_size
    total_bytes = val_bytes.numel() - 1
    all_starts = list(range(0, total_bytes - seq_len, stride))
    my_starts = all_starts[rank::world_size]

    loss_sum = torch.zeros((), device=device, dtype=torch.float64)
    byte_count = torch.zeros((), device=device, dtype=torch.float64)

    model.eval()
    with torch.inference_mode():
        for i in range(0, len(my_starts), batch_size):
            batch_starts = my_starts[i:i + batch_size]
            starts_t = torch.tensor(batch_starts, dtype=torch.int64)
            offsets = torch.arange(seq_len + 1, dtype=torch.int64)
            indices = starts_t.unsqueeze(1) + offsets.unsqueeze(0)
            local_batch = val_bytes[indices].to(device=device, dtype=torch.int64, non_blocking=True)
            x, y = local_batch[:, :-1], local_batch[:, 1:]

            with torch.autocast(device_type="cuda", dtype=torch.bfloat16):
                per_byte_loss = model(x, y, reduction="none", temperature=temperature).detach()

            for b, start in enumerate(batch_starts):
                score_from = 0 if start == 0 else seq_len - stride
                scored = per_byte_loss[b, score_from:]
                loss_sum += scored.to(torch.float64).sum()
                byte_count += scored.numel()

    if dist.is_available() and dist.is_initialized():
        for t in (loss_sum, byte_count):
            dist.all_reduce(t, op=dist.ReduceOp.SUM)

    val_loss = loss_sum / byte_count
    bpb = val_loss.item() / math.log(2.0)
    model.train()
    return float(val_loss.item()), float(bpb)


# ---------------------------------------------------------------------------
# JEPA Diagnostics — verify representation quality independent of BPB
# ---------------------------------------------------------------------------
def jepa_diagnostics(model: nn.Module, val_bytes: Tensor, device: torch.device,
                     seq_len: int = 2048, max_samples: int = 50000):
    """
    Check if JEPA is learning, independent of probability calibration.
    1. Top-k accuracy: is the correct next byte among the k nearest embeddings?
    2. Linear probe: can a simple linear layer decode the next byte from hidden states?
    """
    model.eval()
    base = model.module if hasattr(model, 'module') else model
    base = base._orig_mod if hasattr(base, '_orig_mod') else base

    # Prepare data
    n = min(max_samples + 1, val_bytes.numel())
    data = val_bytes[:n].to(device=device, dtype=torch.int64)
    x = data[:-1].reshape(-1, seq_len)
    y = data[1:].reshape(-1, seq_len)
    n_seqs = min(x.shape[0], max_samples // seq_len)
    x, y = x[:n_seqs], y[:n_seqs]

    all_top1 = 0
    all_top5 = 0
    all_top10 = 0
    all_count = 0
    all_hidden = []
    all_targets = []

    with torch.inference_mode():
        for i in range(n_seqs):
            xi = x[i:i+1]
            yi = y[i:i+1]

            with torch.autocast(device_type="cuda", dtype=torch.bfloat16):
                h = base.encode(xi)                      # (1, T, D)
                pred_emb = base.predictor(h)              # (1, T, D)

            # Distance to all 256 byte embeddings
            all_emb = base.byte_embed.weight              # (256, D)
            # (T, 256) distances
            pred_flat = pred_emb[0].float()
            dist = torch.cdist(pred_flat, all_emb.float().unsqueeze(0)).squeeze(0)  # (T, 256)
            # Nearest neighbors (smallest distance = rank 0)
            ranks = dist.argsort(dim=-1)                  # (T, 256)
            targets = yi[0]                               # (T,)

            # Top-k: is the correct byte in the top k?
            for k, counter_name in [(1, 'top1'), (5, 'top5'), (10, 'top10')]:
                topk = ranks[:, :k]
                hits = (topk == targets.unsqueeze(1)).any(dim=1).sum().item()
                if k == 1: all_top1 += hits
                elif k == 5: all_top5 += hits
                else: all_top10 += hits

            all_count += targets.numel()

            # Collect hidden states for linear probe (limit memory)
            if len(all_hidden) * seq_len < 100000:
                all_hidden.append(h[0].float().cpu())
                all_targets.append(targets.cpu())

    top1_acc = all_top1 / max(all_count, 1)
    top5_acc = all_top5 / max(all_count, 1)
    top10_acc = all_top10 / max(all_count, 1)

    # Linear probe: train a tiny linear layer to predict next byte from hidden states
    probe_loss = float('nan')
    if all_hidden:
        H = torch.cat(all_hidden, dim=0)           # (N, D)
        T = torch.cat(all_targets, dim=0)           # (N,)
        n_train = int(0.8 * H.shape[0])
        H_tr, H_val = H[:n_train].to(device), H[n_train:].to(device)
        T_tr, T_val = T[:n_train].to(device), T[n_train:].to(device)

        d = H.shape[-1]
        probe = nn.Linear(d, 256).to(device)
        opt = torch.optim.Adam(probe.parameters(), lr=1e-3)
        # Quick train: 200 steps
        bs = min(4096, n_train)
        for step in range(200):
            idx = torch.randint(0, n_train, (bs,), device=device)
            logits = probe(H_tr[idx])
            loss = F.cross_entropy(logits, T_tr[idx])
            opt.zero_grad()
            loss.backward()
            opt.step()

        # Eval probe
        with torch.inference_mode():
            val_logits = probe(H_val)
            probe_loss = F.cross_entropy(val_logits, T_val).item()
            probe_bpb = probe_loss / math.log(2.0)

    model.train()
    return {
        "top1_acc": top1_acc,
        "top5_acc": top5_acc,
        "top10_acc": top10_acc,
        "probe_ce": probe_loss,
        "probe_bpb": probe_bpb,
        "n_samples": all_count,
    }

# ---------------------------------------------------------------------------
def find_best_temperature(args, model, rank, world_size, device,
                          grad_accum_steps, val_bytes):
    best_t, best_loss = 1.0, float("inf")
    for t in [0.5, 0.7, 0.85, 0.95, 1.0, 1.1, 1.3, 1.5, 2.0]:
        loss, _ = eval_val(args, model, rank, world_size, device,
                           grad_accum_steps, val_bytes, temperature=t)
        if loss < best_loss:
            best_loss = loss
            best_t = t
    return best_t

# Serialization — FP16 + LZMA
# ---------------------------------------------------------------------------
def save_model(model: nn.Module, code: str, path: str) -> tuple[int, int]:
    """Save model as FP16 + LZMA compressed. Returns (model_bytes, code_bytes)."""
    sd = {k: v.detach().cpu().half() for k, v in model.state_dict().items()}
    buf = io.BytesIO()
    torch.save(sd, buf)
    compressed = lzma.compress(buf.getvalue(), preset=9)
    with open(path, "wb") as f:
        f.write(compressed)
    code_bytes = len(code.encode("utf-8"))
    return len(compressed), code_bytes


def load_model(path: str, model: nn.Module, device: torch.device) -> nn.Module:
    """Load FP16 + LZMA compressed model."""
    with open(path, "rb") as f:
        data = lzma.decompress(f.read())
    sd = torch.load(io.BytesIO(data), map_location="cpu", weights_only=False)
    sd = {k: v.float() for k, v in sd.items()}
    model.load_state_dict(sd, strict=True)
    return model.to(device)

def prepare_byte_data(tokenized_dir: str, tokenizer_path: str, output_dir: str,
                      max_shards: int = 0):
    """Convert tokenized FineWeb shards to raw byte shards."""
    import sentencepiece as spm
    sp = spm.SentencePieceProcessor(model_file=tokenizer_path)
    os.makedirs(output_dir, exist_ok=True)

    shard_files = sorted(glob.glob(os.path.join(tokenized_dir, "fineweb_*.bin")))
    if max_shards > 0:
        # Always include val shards
        val_shards = [f for f in shard_files if "val" in f]
        train_shards = [f for f in shard_files if "train" in f][:max_shards]
        shard_files = val_shards + train_shards

    for shard_file in shard_files:
        header = np.fromfile(shard_file, dtype="<i4", count=256)
        if header.size != 256 or int(header[0]) != 20240520:
            print(f"Skipping {shard_file}: unexpected header")
            continue
        num_tokens = int(header[2])
        header_bytes = 256 * np.dtype("<i4").itemsize
        tokens = np.fromfile(shard_file, dtype="<u2", count=num_tokens, offset=header_bytes)

        # Decode tokens → text → UTF-8 bytes
        text = sp.decode(tokens.tolist())
        raw_bytes = np.frombuffer(text.encode("utf-8"), dtype=np.uint8)

        out_name = os.path.basename(shard_file)
        out_path = os.path.join(output_dir, out_name)
        raw_bytes.tofile(out_path)
        print(f"  {shard_file} → {out_path} ({len(raw_bytes):,} bytes)")

SCALAR_PARAM_PATTERNS = ("log_temperature", "bias")


def main() -> None:
    # Handle --prepare-data mode
    if "--prepare-data" in sys.argv:
        import argparse
        parser = argparse.ArgumentParser()
        parser.add_argument("--prepare-data", action="store_true")
        parser.add_argument("--tokenized-dir", required=True)
        parser.add_argument("--tokenizer-path", required=True)
        parser.add_argument("--output-dir", required=True)
        parser.add_argument("--max-shards", type=int, default=0)
        pargs = parser.parse_args()
        prepare_byte_data(pargs.tokenized_dir, pargs.tokenizer_path,
                          pargs.output_dir, pargs.max_shards)
        return

    args = Hyperparameters()
    code = Path(__file__).read_text(encoding="utf-8")

    global ns_orth
    ns_orth = torch.compile(_ns_orth)

    # --- Distributed setup ---
    distributed = "RANK" in os.environ and "WORLD_SIZE" in os.environ
    rank = int(os.environ.get("RANK", "0"))
    world_size = int(os.environ.get("WORLD_SIZE", "1"))
    local_rank = int(os.environ.get("LOCAL_RANK", "0"))
    grad_accum_steps = max(1, 8 // world_size)
    grad_scale = 1.0 / grad_accum_steps

    if not torch.cuda.is_available():
        raise RuntimeError("CUDA required")
    device = torch.device("cuda", local_rank)
    torch.cuda.set_device(device)
    if distributed:
        dist.init_process_group(backend="nccl", device_id=device)
        dist.barrier()
    master = rank == 0
    torch.backends.cuda.matmul.allow_tf32 = True
    torch.backends.cudnn.allow_tf32 = True

    os.makedirs("logs/cuda/", exist_ok=True)
    logfile = f"logs/cuda/{args.run_id}.txt" if master else None

    def log0(msg: str, console: bool = True):
        if not master:
            return
        if console:
            print(msg)
        if logfile:
            with open(logfile, "a", encoding="utf-8") as f:
                print(msg, file=f)

    log0(code, console=False)
    log0("=" * 100, console=False)

    random.seed(args.seed)
    np.random.seed(args.seed)
    torch.manual_seed(args.seed)
    torch.cuda.manual_seed_all(args.seed)

    # --- Validation data ---
    val_bytes = load_val_bytes(args.val_files, args.train_seq_len)

    # --- Model ---
    base_model = ByteJEPA(
        model_dim=args.model_dim,
        num_layers=args.num_layers,
        d_state=args.d_state,
        d_conv=args.d_conv,
        expand=args.expand,
        predictor_hidden_mult=args.predictor_hidden_mult,
    ).to(device).float()

    sigreg = SIGReg(knots=args.sigreg_knots, num_proj=args.sigreg_num_proj).to(device)

    torch._dynamo.config.optimize_ddp = False
    compiled_model = torch.compile(base_model,
                                   mode=args.compile_mode if args.compile_mode != "default" else None)
    model = DDP(compiled_model, device_ids=[local_rank],
                broadcast_buffers=False, gradient_as_bucket_view=True) if distributed else compiled_model

    # --- Optimizers ---
    matrix_params = [p for n, p in base_model.named_parameters()
                     if p.ndim == 2 and not any(pat in n for pat in SCALAR_PARAM_PATTERNS)]
    scalar_params = [p for n, p in base_model.named_parameters()
                     if p.ndim < 2 or any(pat in n for pat in SCALAR_PARAM_PATTERNS)]
    embed_params = [base_model.byte_embed.weight]
    # Remove embed from matrix_params to avoid double-counting
    embed_ids = {id(p) for p in embed_params}
    matrix_params = [p for p in matrix_params if id(p) not in embed_ids]

    opt_muon = Muon(matrix_params, lr=args.matrix_lr, momentum=args.muon_momentum,
                    backend_steps=args.muon_backend_steps)
    for g in opt_muon.param_groups:
        g["base_lr"] = args.matrix_lr

    opt_scalar = torch.optim.Adam(
        [{"params": scalar_params, "lr": args.scalar_lr, "base_lr": args.scalar_lr}],
        betas=(args.beta1, args.beta2), eps=args.adam_eps, fused=True)

    opt_embed = torch.optim.Adam(
        [{"params": embed_params, "lr": args.embed_lr, "base_lr": args.embed_lr}],
        betas=(args.beta1, args.beta2), eps=args.adam_eps, fused=True)

    optimizers = [opt_muon, opt_scalar, opt_embed]

    # --- Log config ---
    n_params = sum(p.numel() for p in base_model.parameters())
    log0(f"ByteJEPA | params:{n_params:,} layers:{args.num_layers} dim:{args.model_dim} "
         f"d_state:{args.d_state} expand:{args.expand} | "
         f"ssm_backend:{'mamba2_cuda' if MAMBA_AVAILABLE else 'mamba2_pytorch'} "
         f"ws:{world_size} ga:{grad_accum_steps}")
    log0(f"JEPA: jepa_weight={args.jepa_weight} sigreg_lambda={args.sigreg_lambda} ce_weight={args.ce_weight} detach_targets={args.detach_targets}")

    # --- Data loader ---
    train_loader = DistributedByteLoader(args.train_files, rank, world_size, device)

    def zero_grad_all():
        for opt in optimizers:
            opt.zero_grad(set_to_none=True)

    max_wallclock_ms = 1000.0 * args.max_wallclock_seconds if args.max_wallclock_seconds > 0 else None

    def lr_mul(step: int, elapsed_ms: float):
        if args.warmdown_fraction <= 0:
            return 1.0
        if max_wallclock_ms is None:
            wd_start = int(args.iterations * (1.0 - args.warmdown_fraction))
            if step < wd_start:
                return 1.0
            return max((args.iterations - step) / max(args.iterations * args.warmdown_fraction, 1), 0.0)
        wd_ms = max_wallclock_ms * args.warmdown_fraction
        remaining = max(max_wallclock_ms - elapsed_ms, 0.0)
        return remaining / max(wd_ms, 1e-9) if remaining <= wd_ms else 1.0

    # --- Main training loop ---
    training_time_ms = 0.0
    stop_after_step: int | None = None
    train_loss = torch.zeros((), device=device)
    torch.cuda.synchronize()
    t0 = time.perf_counter()
    step = 0

    while True:
        last_step = step == args.iterations or (stop_after_step is not None and step >= stop_after_step)

        # Validation
        if last_step or (args.val_loss_every > 0 and step % args.val_loss_every == 0):
            torch.cuda.synchronize()
            training_time_ms += 1000.0 * (time.perf_counter() - t0)
            val_loss, val_bpb = eval_val(args, model, rank, world_size, device,
                                         grad_accum_steps, val_bytes)
            log0(f"step:{step}/{args.iterations} val_loss:{val_loss:.4f} val_bpb:{val_bpb:.4f} "
                 f"train_time:{training_time_ms:.0f}ms")
            torch.cuda.synchronize()
            t0 = time.perf_counter()

        if last_step:
            break

        # LR scheduling
        elapsed_ms = training_time_ms + 1000.0 * (time.perf_counter() - t0)
        scale = lr_mul(step, elapsed_ms)

        # Muon momentum warmup
        if args.muon_momentum_warmup_steps > 0:
            frac = min(step / args.muon_momentum_warmup_steps, 1.0)
            for g in opt_muon.param_groups:
                g["momentum"] = ((1 - frac) * args.muon_momentum_warmup_start
                                 + frac * args.muon_momentum)

        # Training step
        zero_grad_all()
        train_loss.zero_()

        for micro in range(grad_accum_steps):
            if distributed:
                model.require_backward_grad_sync = micro == grad_accum_steps - 1
            x, y = train_loader.next_batch(args.train_batch_tokens, args.train_seq_len,
                                           grad_accum_steps)
            torch.compiler.cudagraph_mark_step_begin()
            with torch.autocast(device_type="cuda", dtype=torch.bfloat16):
                loss, diag = base_model.compute_jepa_loss(
                    x, y, sigreg, args.sigreg_lambda, args.ce_weight,
                    args.jepa_weight, args.detach_targets)
            train_loss.add_(loss.detach())
            (loss * grad_scale).backward()

        train_loss /= grad_accum_steps

        # Optimizer step with LR scaling
        for opt in optimizers:
            for g in opt.param_groups:
                g["lr"] = g["base_lr"] * scale
            opt.step()
        zero_grad_all()

        step += 1
        approx_ms = training_time_ms + 1000.0 * (time.perf_counter() - t0)

        if args.train_log_every > 0 and step % args.train_log_every == 0:
            d_str = " ".join(f"{k}:{v.item():.4f}" for k, v in diag.items())
            log0(f"step:{step}/{args.iterations} loss:{train_loss.item():.4f} "
                 f"{d_str} t:{approx_ms:.0f}ms")

        # Wallclock cap
        if stop_after_step is None and max_wallclock_ms is not None and step % 10 == 0:
            reached = approx_ms >= max_wallclock_ms
            if distributed:
                cap_t = torch.tensor(int(reached), device=device)
                dist.all_reduce(cap_t, op=dist.ReduceOp.MAX)
                reached = bool(cap_t.item())
            if reached:
                stop_after_step = step

    # --- Serialization ---
    if master:
        model_bytes, code_bytes = save_model(base_model, code, "final_model.fp16.ptz")
        total = model_bytes + code_bytes
        log0(f"artifact:{model_bytes / 1e6:.2f}MB code:{code_bytes} "
             f"total:{total}/{16_000_000} ({total / 1e6:.2f}/16.00MB) "
             f"{'FITS' if total <= 16_000_000 else 'OVER'}")

    # --- Roundtrip evaluation ---
    if distributed:
        dist.barrier()

    load_model("final_model.fp16.ptz", base_model, device)
    torch._dynamo.reset()

    q_loss, q_bpb = eval_val(args, model, rank, world_size, device,
                              grad_accum_steps, val_bytes)
    log0(f"final_roundtrip val_loss:{q_loss:.4f} val_bpb:{q_bpb:.4f}")

    # JEPA representation diagnostics
    if master:
        diag = jepa_diagnostics(model, val_bytes, device, seq_len=args.train_seq_len)
        log0(f"jepa_diagnostics top1:{diag['top1_acc']:.4f} top5:{diag['top5_acc']:.4f} "
             f"top10:{diag['top10_acc']:.4f} "
             f"linear_probe_ce:{diag['probe_ce']:.4f} linear_probe_bpb:{diag['probe_bpb']:.4f}")

    # Temperature search
    if args.temp_scaling:
        torch.cuda.synchronize()
        t_temp = time.perf_counter()
        opt_temp = find_best_temperature(args, base_model, rank, world_size,
                                         device, grad_accum_steps, val_bytes)
        torch.cuda.synchronize()
        log0(f"temp_scaling optimal_T:{opt_temp:.2f} "
             f"time:{1000.0 * (time.perf_counter() - t_temp):.0f}ms")

        # Re-eval with optimal temperature
        t_loss, t_bpb = eval_val(args, model, rank, world_size, device,
                                  grad_accum_steps, val_bytes, temperature=opt_temp)
        log0(f"final_temped val_loss:{t_loss:.4f} val_bpb:{t_bpb:.4f} T={opt_temp:.2f}")

    # Sliding window eval
    if args.sliding_eval:
        torch.cuda.synchronize()
        t_sw = time.perf_counter()
        temp = opt_temp if args.temp_scaling else 1.0
        sw_loss, sw_bpb = eval_val_sliding(args, base_model, rank, world_size,
                                            device, grad_accum_steps, val_bytes,
                                            stride=args.sliding_eval_stride,
                                            temperature=temp)
        torch.cuda.synchronize()
        log0(f"final_sliding val_loss:{sw_loss:.4f} val_bpb:{sw_bpb:.4f} "
             f"(stride={args.sliding_eval_stride}, T={temp:.2f}) "
             f"time:{1000.0 * (time.perf_counter() - t_sw):.0f}ms")

    if distributed:
        dist.destroy_process_group()


if __name__ == "__main__":
    main()
